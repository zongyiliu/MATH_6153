		\section{Question 1}
		
		Let $\left(M_{n}\right)_{n \ge 0}$ be a discrete-time martingale in 
		$\left(\Omega, \mathcal{F}, (\mathcal{F}_{n})_{n \ge 0}, \mathbb{P}\right)$. 
		Assume $M_{0}=0$, $\mathbb{E}(M_{n}^{4})<\infty$ for all $n \ge 0$, and write
		$\Delta_{n}=M_{n}-M_{n-1}$ for $n \ge 1$.
		
		\begin{enumerate}
			\item[a.] Show that there are absolute constants $c, C>0$ 
			(i.e.\ not depending on $M$ or $n$) such that
			$$
			c\,\mathbb{E}\!\left((\Delta_{1}^{2}+\cdots+\Delta_{n}^{2})^{2}\right)
			\le
			\mathbb{E}\!\left(M_{n}^{4}\right)
			\le
			C\,\mathbb{E}\!\left((\Delta_{1}^{2}+\cdots+\Delta_{n}^{2})^{2}\right).
			$$
			
			\item[b.] Let $(H_n)$ be an $(\mathcal{F}_n)$-adapted process such that
			$|H_n|\le 1$ for all $n\ge 0$. Let $(N_n)$ be such that $N_0=0$ and
			$$
			N_n - N_{n-1} = H_{n-1}\Delta_n, \qquad n \ge 1.
			$$
			Show that there is an absolute constant $C'$ (i.e.\ not depending on $M$, $H$
			or $n$) such that
			$$
			\mathbb{E}(N_n^{4}) \le C'\,\mathbb{E}(M_n^{4}).
			$$
		\end{enumerate}
		
		\textbf{Answer}

		\section{Question 2}
		
		Consider the branching process $(X_n)_{n \ge 0}$ taking values in $\mathbb{N}$ given by
		$X_0 = x_0$ and
		$$
		X_{n+1} = \sum_{m=1}^{X_n} \xi_{m,n},
		$$
		where the $(\xi_{m,n})_{m,n \ge 1}$ are iid with common distribution $\mu$
		(and values in $\mathbb{N}$). Set
		$$
		m = \sum_{k \ge 0} k\,\mu\{k\}.
		$$
		
		\begin{enumerate}
			\item[a.] Show that $(X_n/m^n)_{n \ge 0}$ is a martingale
			(in the canonical filtration of $(X_n)$).
			
			\item[b.] Show that if $m \le 1$, a.s.\ $(X_n)$ reaches eventually
			the absorbing state $0$, except in the trivial case $\mu=\delta_1$.
			(Use martingale convergence).
			
			\item[c.] Assume now that $m>1$ and $\sigma^2=\operatorname{Var}(\xi)<\infty$.
			Compute $\mathbb{E}(X_{n+1}^2)$ in terms of $\mathbb{E}(X_n^2)$, $m$, and $\sigma$.
			Conclude that $(X_n/m^n)$ is bounded in $L^2$ and that
			$$
			\lim X_n = \infty
			$$
			with positive probability.
		\end{enumerate}
		
		\textbf{Answer}
		
		\clearpage
